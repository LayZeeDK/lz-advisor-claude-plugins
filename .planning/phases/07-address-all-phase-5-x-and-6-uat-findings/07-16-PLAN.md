    ```javascript
    import { readFileSync } from 'node:fs';
    const body = readFileSync(process.argv[2], 'utf8');
    // Plan 07-10 fragment-grammar items: numbered list, may have **Assuming X (unverified):** markers
    const ITEM_RE = /^\d+\.\s+(.+?)(?=\n\d+\.\s+|\n\*\*Critical|\n*$)/gms;
    const matches = [...body.matchAll(ITEM_RE)];
    if (matches.length === 0) {
      console.log('[ERROR] No numbered items detected');
      process.exit(1);
    }
    matches.forEach((m, idx) => {
      const txt = m[1].trim();
      const wc = txt.split(/\s+/).filter(Boolean).length;
      const hasAssuming = /Assuming.+\(unverified\)/i.test(txt);
      const cap = hasAssuming ? 22 : 15;
      const tag = wc > cap ? '[OVER]' : '[OK]';
      console.log(`${tag} item ${idx + 1}: ${wc}w (cap ${cap}w${hasAssuming ? ', Assuming-frame' : ''})`);
    });
    ```
